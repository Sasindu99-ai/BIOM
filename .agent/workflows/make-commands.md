# Sync dependencies: make sync

# Check code style: make lint

# Fix code style: make fix

# Run vite dev server: make dev

# Build the project: make build

# Run Django production server: make run

# Make migrations: make make-migrations

# Execute migrations: make execute-migrate

# Make and execute migrations: make migrate

# Create a superuser: make superuser

# Django shell: make shell

# Django DB shell: make db

# Update project dependencies + make migrations + migrate: make update
